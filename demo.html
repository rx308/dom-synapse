DomSynapse Demo

Архитектура: Обработчики привязываются к конкретным типам событий.

Метод connect('имя', callback, 'тип_события') позволяет указать тип события.

Базовое использование (клики)

Обработчики для событий клика

// Обработчики для кликов
synapse
    .connect('alert', (element, event, data) => {
        alert(`Сработал элемент: ${data.message || element.textContent}`);
    }, 'click')
    .connect('log', (element, event, data) => {
        logEvent(`Клик на: ${data.message || element.textContent}`);
    }, 'click')
    .connect('counter', (element, event, data) => {
        const counter = document.getElementById('counter');
        counter.textContent = parseInt(counter.textContent) + 1;
    }, 'click');

// Наблюдение за кликами
synapse.observe(document, ['click']);

Разные типы событий

Обработчики для различных типов событий

// Обработчики для разных событий
synapse
    .connect('hover', (element, event, data) => {
        logEvent(`Наведение на: ${data.message || element.textContent}`);
    }, 'mouseover')
    .connect('key', (element, event, data) => {
        logEvent(`Нажата клавиша: ${event.key} на: ${data.message || element.textContent}`);
    }, 'keydown')
    .connect('mousedown', (element, event, data) => {
        logEvent(`MouseDown на: ${data.message || element.textContent}`);
    }, 'mousedown')
    .connect('input', (element, event, data) => {
        logEvent(`Ввод в поле: ${element.value}`);
    }, 'input');

// Наблюдение за дополнительными событиями
synapse.observeEvent('mouseover');
synapse.observeEvent('keydown');
synapse.observeEvent('mousedown');
synapse.observeEvent('input');

Автозагрузка

Элементы с атрибутом data-synapse-load автоматически запускают обработчики при загрузке

// Обработчик для автозагрузки
synapse.connect('autoload', (element, event, data) => {
    logEvent(`Автозагрузка: ${data.message || element.textContent}`);
}, 'load');
Этот элемент автоматически запустит обработчик при загрузке страницы
